package sandya;

import java.util.Scanner;

public class occurenvce 
     {
	public static void main(String[] args) {
		int no,digit,m,oc=0;
		Scanner s=new Scanner(System.in);
		System.out.println("Enetr the number=");
		no=s.nextInt();
		System.out.println("Enter the digit=");
        digit=s.nextInt();
        while(no>0)
        {
        	m=no%10;
        	if(m==digit)
        		oc++;
        	no=no/10;
        	
        }
        System.out.println("Digit occured "+oc+" times.");
        }
   }
